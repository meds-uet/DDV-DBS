module decoder(
input logic[3:0] a;
output logic [6:0] y);
always@(*) begin
case(a)
4'b0000 : y=1111110;
4'b0001 : y=0110000;
4'b0010 : y=1101101;
4'b0011 : y=1111001;
4'b0100 : y=0110011;
4'b0101 : y=1011011;
4'b0110 : y=1011111;
4'b0111 : y=1110000;
4'b1000 : y=1111111;
4'b1001 : y=1111011;
4'b1010 : y=0110011;
4'b1011 : y=1011011;
4'b1100 : y=1011111;
4'b1101 : y=1110000;
4'b1110 : y=1111111;
4'b1111 : y=1111011;
endcase
end
endmodule
